18/**
19 * Make a list of constraints.
20 *
21 * The standard form for equality constraints is c(x) = 0, and the standard form
22 * for inequality constraints is c(x) ≥ 0. This function takes constraints of
23 * the form lhs = rhs or lhs ≥ rhs and converts them to lhs - rhs = 0 or
24 * lhs - rhs ≥ 0.
25 *
26 * @param lhs Left-hand side.
27 * @param rhs Right-hand side.
28 */
29template <typename LHS, typename RHS>
30 requires(ScalarLike<std::decay_t<LHS>> || MatrixLike<std::decay_t<LHS>>) &&
31 (ScalarLike<std::decay_t<RHS>> || MatrixLike<std::decay_t<RHS>>) &&
32 (!std::same_as<std::decay_t<LHS>, double> ||
33 !std::same_as<std::decay_t<RHS>, double>)
34wpi::SmallVector<Variable> MakeConstraints(LHS&& lhs, RHS&& rhs) {
35 wpi::SmallVector<Variable> constraints;
36
37 if constexpr (ScalarLike<std::decay_t<LHS>> &&
38 ScalarLike<std::decay_t<RHS>>) {
39 constraints.emplace_back(lhs - rhs);
40 } else if constexpr (ScalarLike<std::decay_t<LHS>> &&
41 MatrixLike<std::decay_t<RHS>>) {
42 int rows;
43 int cols;
44 if constexpr (EigenMatrixLike<std::decay_t<RHS>>) {
45 rows = rhs.rows();
46 cols = rhs.cols();
47 } else {
48 rows = rhs.Rows();
49 cols = rhs.Cols();
50 }
51
52 constraints.reserve(rows * cols);
53
54 for (int row = 0; row < rows; ++row) {
55 for (int col = 0; col < cols; ++col) {
56 // Make right-hand side zero
57 constraints.emplace_back(lhs - rhs(row, col));
58 }
59 }
60 } else if constexpr (MatrixLike<std::decay_t<LHS>> &&
61 ScalarLike<std::decay_t<RHS>>) {
62 int rows;
63 int cols;
64 if constexpr (EigenMatrixLike<std::decay_t<LHS>>) {
65 rows = lhs.rows();
66 cols = lhs.cols();
67 } else {
68 rows = lhs.Rows();
69 cols = lhs.Cols();
70 }
71
72 constraints.reserve(rows * cols);
73
74 for (int row = 0; row < rows; ++row) {
75 for (int col = 0; col < cols; ++col) {
76 // Make right-hand side zero
77 constraints.emplace_back(lhs(row, col) - rhs);
78 }
79 }
80 } else if constexpr (MatrixLike<std::decay_t<LHS>> &&
81 MatrixLike<std::decay_t<RHS>>) {
82 int lhsRows;
83 int lhsCols;
84 if constexpr (EigenMatrixLike<std::decay_t<LHS>>) {
85 lhsRows = lhs.rows();
86 lhsCols = lhs.cols();
87 } else {
88 lhsRows = lhs.Rows();
89 lhsCols = lhs.Cols();
90 }
91
92 [[maybe_unused]]
93 int rhsRows;
94 [[maybe_unused]]
95 int rhsCols;
96 if constexpr (EigenMatrixLike<std::decay_t<RHS>>) {
97 rhsRows = rhs.rows();
98 rhsCols = rhs.cols();
99 } else {
100 rhsRows = rhs.Rows();
101 rhsCols = rhs.Cols();
102 }
103
104 Assert(lhsRows == rhsRows && lhsCols == rhsCols);
105 constraints.reserve(lhsRows * lhsCols);
106
107 for (int row = 0; row < lhsRows; ++row) {
108 for (int col = 0; col < lhsCols; ++col) {
109 // Make right-hand side zero
110 constraints.emplace_back(lhs(row, col) - rhs(row, col));
111 }
112 }
113 }
114
115 return constraints;
116}

118/**
119 * A vector of equality constraints of the form cₑ(x) = 0.
120 */
121struct SLEIPNIR_DLLEXPORT EqualityConstraints {
122 /// A vector of scalar equality constraints.
123 wpi::SmallVector<Variable> constraints;
124
125 /**
126 * Constructs an equality constraint from a left and right side.
127 *
128 * The standard form for equality constraints is c(x) = 0. This function takes
129 * a constraint of the form lhs = rhs and converts it to lhs - rhs = 0.
130 *
131 * @param lhs Left-hand side.
132 * @param rhs Right-hand side.
133 */
134 template <typename LHS, typename RHS>
135 requires(ScalarLike<std::decay_t<LHS>> || MatrixLike<std::decay_t<LHS>>) &&
136 (ScalarLike<std::decay_t<RHS>> || MatrixLike<std::decay_t<RHS>>) &&
137 (!std::same_as<std::decay_t<LHS>, double> ||
138 !std::same_as<std::decay_t<RHS>, double>)
139 EqualityConstraints(LHS&& lhs, RHS&& rhs)
140 : constraints{MakeConstraints(lhs, rhs)} {}
141
142 /**
143 * Implicit conversion operator to bool.
144 */
145 operator bool() { // NOLINT
146 return std::all_of(
147 constraints.begin(), constraints.end(),
148 [](auto& constraint) { return constraint.Value() == 0.0; });
149 }
150};
